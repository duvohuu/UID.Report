\chapter{MỘT SỐ QUY UỚC}
    \section{Ngôn ngữ lập trình}
        \begin{itemize}
            \item Ngôn ngữ lập trình chính được sử dụng trong codebase là JavaScript với chuẩn ES6. 
            \item Front-end sử dụng JavaScript với framework ReactJS.
            \item Back-end sử dụng JavaScript trong môi truờng Node.js.
        \end{itemize}
    \section{Package manager}
        \begin{itemize}
            \item Package manager được sử dụng trong dự án là \textbf{yarn} thay vì \textbf{npm} nhằm cải thiện hiệu suất cài đặt và quản lý các gói.
            \item Tất cả các thư viện bên thứ ba được quản lý thông qua file \texttt{package.json} ở thư mục gốc của dự án.
        \end{itemize}
